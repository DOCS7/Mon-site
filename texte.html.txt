<script>
    function showTab(tab) {
        document.querySelectorAll('.content').forEach(el => el.classList.remove('active'));
        document.querySelectorAll('.tab').forEach(el => el.classList.remove('active'));
        document.getElementById(tab).classList.add('active');
        document.querySelector(`.tab[onclick*="'${tab}'"]`).classList.add('active');
    }

    function openForm(category, product) {
        document.getElementById('selectedProduct').innerHTML = 'Produit sélectionné : <strong>' + product + '</strong>';
        document.getElementById('orderForm').classList.add('show');
        document.getElementById('orderStatus').classList.remove('show');
        window.scrollTo({ top: document.getElementById('orderForm').offsetTop - 100, behavior: 'smooth' });
    }

    function closeForm() {
        document.getElementById('orderForm').classList.remove('show');
    }

    function submitOrder() {
        const lastName = document.getElementById('lastName').value.trim();
        const firstName = document.getElementById('firstName').value.trim();
        const email = document.getElementById('email').value.trim();
        const address = document.getElementById('address').value.trim();
        const agree = document.getElementById('agree').checked;

        if (!lastName || !firstName || !email || !address) {
            alert('Veuillez remplir tous les champs obligatoires (*).');
            return;
        }
        if (!agree) {
            alert('Veuillez accepter les conditions de traitement des données.');
            return;
        }

        const ref = 'DOC-2026-' + Math.floor(1000 + Math.random() * 9000);
        document.getElementById('orderRef').textContent = ref;
        document.getElementById('statusMessage').innerHTML =
            'Merci ' + firstName + ' ! Votre commande a bien été reçue.<br>' +
            'Un email de confirmation sera envoyé à <strong>' + email + '</strong> sous 24h avec les instructions de paiement.';

        document.getElementById('orderForm').classList.remove('show');
        document.getElementById('orderStatus').classList.add('show');
        window.scrollTo({ top: 0, behavior: 'smooth' });
    }

    function resetPage() {
        document.getElementById('orderStatus').classList.remove('show');
        document.getElementById('orderForm').classList.remove('show');
        document.querySelectorAll('#orderForm input, #orderForm textarea, #orderForm select').forEach(el => {
            if (el.type !== 'checkbox') el.value = '';
            else el.checked = false;
        });
    }
</script>
